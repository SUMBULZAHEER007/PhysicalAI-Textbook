# Module 4: Robotic Nervous System - ROS 2

## Overview

ROS 2 (Robot Operating System 2) is the middleware for building robot applications. It provides communication, hardware abstraction, and standard algorithms.

### ROS 2 Core Concepts

1. **Nodes**: Independent processes
2. **Topics**: Asynchronous publish/subscribe
3. **Services**: Synchronous request/response
4. **Actions**: Long-running goal-oriented tasks

---

## Section 1: Nodes and Topics

### Creating a Publisher

```python
import rclpy
from rclpy.node import Node
from std_msgs.msg import Float32

class JointPublisher(Node):
    def __init__(self):
        super().__init__('joint_publisher')
        self.publisher = self.create_publisher(Float32, 'joint_angle', 10)
        self.timer = self.create_timer(0.1, self.publish_joint)
    
    def publish_joint(self):
        msg = Float32()
        msg.data = 45.0  # degrees
        self.publisher.publish(msg)
        self.get_logger().info(f'Publishing: {msg.data}')

def main(args=None):
    rclpy.init(args=args)
    publisher = JointPublisher()
    rclpy.spin(publisher)
```

### Creating a Subscriber

```python
import rclpy
from rclpy.node import Node
from std_msgs.msg import Float32

class JointSubscriber(Node):
    def __init__(self):
        super().__init__('joint_subscriber')
        self.subscription = self.create_subscription(
            Float32, 'joint_angle', self.listener_callback, 10
        )
    
    def listener_callback(self, msg):
        self.get_logger().info(f'Received angle: {msg.data}°')

def main(args=None):
    rclpy.init(args=args)
    subscriber = JointSubscriber()
    rclpy.spin(subscriber)
```

## Section 2: Services

### Define a Service

```python
from srv_interfaces.srv import SetJointAngle

class JointController(Node):
    def __init__(self):
        super().__init__('joint_controller')
        self.srv = self.create_service(
            SetJointAngle, 'set_angle', self.handle_set_angle
        )
    
    def handle_set_angle(self, request, response):
        angle = request.angle
        self.get_logger().info(f'Setting joint to {angle}°')
        response.success = True
        return response
```

### Call a Service

```python
from srv_interfaces.srv import SetJointAngle

class AngleController(Node):
    def __init__(self):
        super().__init__('angle_controller')
        self.cli = self.create_client(SetJointAngle, 'set_angle')
    
    def call_service(self, angle_value):
        while not self.cli.wait_for_service(timeout_sec=1.0):
            self.get_logger().info('Service not available...')
        
        req = SetJointAngle.Request()
        req.angle = angle_value
        self.future = self.cli.call_async(req)
```

## Section 3: Launch Files

### Python Launch File

```python
from launch import LaunchDescription
from launch_ros.actions import Node

def generate_launch_description():
    return LaunchDescription([
        Node(
            package='joint_control',
            executable='joint_publisher',
            name='publisher',
            output='screen'
        ),
        Node(
            package='joint_control',
            executable='joint_subscriber',
            name='subscriber',
            output='screen'
        ),
    ])
```
